    even_palindrome_count = 0

                odd_palindrome_count += 1

def is_sorted(lst: List[int]) -> bool:

def sort_array(array: List[int]) -> List[int]:
    return [] if len(array) == 0 else sorted(array, reverse= (array[0]+array[-1]) % 2 == 0)
